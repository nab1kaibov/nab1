import java.util.PriorityQueue;
import java.util.HashMap;

public class JavaExample {
    public static void main(String[] args) {
        // Приоритетная очередь (минимальная куча)
        PriorityQueue<Integer> pq = new PriorityQueue<>(); // создаём приоритетную очередь
        pq.add(5);
        pq.add(2);
        pq.add(8);
        System.out.println(pq.poll()); // извлекаем и выводим наименьший элемент (2)
        System.out.println(pq.poll()); // извлекаем следующий наименьший (5)

        // Хеш-таблица (HashMap)
        HashMap<String, Integer> map = new HashMap<>();
        map.put("one", 1);
        map.put("two", 2);
        System.out.println(map.get("one")); // доступ по ключу: 1
        if (map.containsKey("two")) {
            System.out.println("Key two is present"); // проверяем наличие ключа
        }
        map.remove("one"); // удаляем запись с ключом "one"
        System.out.println(map.containsKey("one")); // false – "one" больше нет в карте
    }
}
